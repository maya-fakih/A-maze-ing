ðŸ“Œ BASIC GIT COMMANDS
1. See Where You Are
bash
# See current branch
git branch          # List all branches, * marks current
git status          # See what's changed/staged
git log --oneline   # Recent commits
2. Create & Switch Branches
bash
# Create AND switch to new branch
git checkout -b branch-name

# OR (newer syntax)
git switch -c branch-name

# Just create branch (don't switch)
git branch branch-name

# Switch to existing branch
git checkout branch-name
git switch branch-name

# Switch back to previous branch
git checkout -
git switch -
3. Working with Changes
bash
# Stage files
git add file.py              # Add specific file
git add .                    # Add ALL changes
git add -A                   # Add ALL including deletions

# Commit
git commit -m "Your message"
git commit -am "Message"     # Add and commit tracked files

# See what you changed
git diff                     # Unstaged changes
git diff --staged            # Staged changes

ðŸ”¥ COMMON SCENARIOS
"I messed up, want fresh start"
bash
# Undo uncommitted changes
git checkout -- file.py          # Discard changes to file
git reset --hard HEAD           # Discard ALL uncommitted changes

# Delete local branch
git branch -D branch-name

# Reset to remote (OVERRITES LOCAL!)
git fetch origin
git reset --hard origin/main

bash

# OR

ðŸš¨ EMERGENCY COMMANDS
bash
# Undo last commit (but keep changes)
git reset --soft HEAD~1

# Undo last commit (lose changes)
git reset --hard HEAD~1

# Revert a specific commit
git revert COMMIT_HASH

# Stash changes (save for later)
git stash
git stash pop

# See who wrote what
git blame file.py
ðŸŽ¯ QUICK REFERENCE CARDS
Card 1: Daily Workflow
bash
# Start of day:
git checkout main
git pull origin main
git checkout -b yourname/feature

# During work:
git add .
git commit -m "Message"

# End of day:

# Create PR on GitHub
Card 2: Fixing Mistakes
bash
# Wrong branch?
git stash                     # Save work
git checkout correct-branch
git stash pop                # Restore work

# Wrong commit message?
git commit --amend -m "New message"

# Accidentally committed to main?
git checkout -b fix-branch   # Move commit to new branch
git checkout main
git reset --hard HEAD~1      # Remove from main
Card 3: Seeing What Happened
bash
# Pretty log
git log --oneline --graph --all

# What changed in a file?
git diff HEAD~1 file.py

# Who changed what?
git log -- file.py

# When was line 50 changed?
git blame -L 50,50 file.py
